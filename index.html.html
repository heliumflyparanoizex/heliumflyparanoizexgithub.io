            // Función para animar el texto como si se estuviera escribiendo

            function typeWriter(text, element, onComplete) {

                let i = 0;

                element.innerHTML = "";

                const speed = 40; // Velocidad en milisegundos

               

                function type() {

                    if (i < text.length) {

                        // Añade el efecto de "[Escribiendo...]"

                        if (text.substring(i, i+15) === "Firma vibracional") {

                             element.innerHTML = text.substring(0, i) + '<span class="typing-effect">[Escribiendo...]</span>';

                        } else {

                             element.innerHTML = text.substring(0, i) + '<span class="typing-effect"></span>';

                        }

                       

                        if (i > 1 && text.substring(i-1, i+14) === "Firma vibracional") {

                            element.innerHTML = text.substring(0, i);

                        }

                       

                        i++;

                        setTimeout(type, speed);

                    } else {

                        element.innerHTML = text; // Muestra el texto completo al final

                        if (onComplete) onComplete();

                    }

                }

                type();

            }
